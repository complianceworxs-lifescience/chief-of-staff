def check_agent_health(agent_log):
    if "timestamp" not in agent_log or time_since(agent_log["timestamp"]) > 48 hours:
        return f"{agent_log['agent']} is inactive"
    if "error" in agent_log:
        return f"{agent_log['agent']} reported error: {agent_log['error']}"
    return "healthy"

def detect_conflict(cro_output, cmo_output):
    if "price" in cro_output["recommendations"] and "price" in cmo_output["recommendations"]:
        if cro_output["recommendations"]["price"] != cmo_output["recommendations"]["price"]:
            return True
    return False

def resolve_conflict(cro_output, cmo_output, ceo_policy):
    # Default to strategic priority or defer to CEO
    if ceo_policy.get("growth_mode") and cro_output["recommended_price"] > cmo_output["recommended_price"]:
        return cro_output
    else:
        return {"resolution": "Escalate to CEO agent for arbitration"}

def check_alignment(output, strategy_objectives):
    aligned = []
    for goal in strategy_objectives["q3_goals"]:
        if goal.lower() in output["summary"].lower():
            aligned.append(goal)
    return aligned
