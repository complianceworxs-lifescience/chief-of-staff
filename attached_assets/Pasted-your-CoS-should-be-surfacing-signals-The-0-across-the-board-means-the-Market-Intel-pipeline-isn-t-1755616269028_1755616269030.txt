Backend orchestrator

// /api/mi/ingest-and-score.ts
export default async function handler(req, res) {
  const cfg = await MiConfig.get();                 // keywords, sources, competitors
  if (!cfg || cfg.empty) return res.json({ ok:false, reason:"NO_CONFIG" });

  const raw = await MiCollectors.runAll(cfg);       // fetch feeds/scrapers/APIs
  const scored = await MiScorer.score(raw, cfg);    // priority + category tags
  await MiStore.upsert(scored);                     // writes to the SAME store the dashboard reads

  const stats = await MiStore.stats();              // { total, high_priority, processed_today, assignments }
  return res.json({ ok:true, stats });
}
